SELECT TOP 20 PERCENT	T.TERM_CD, 
							T.TERM_SEQ_NUM, 
							T.TERM_NAME
	FROM (SELECT	TERM_CD, 
					TERM_SEQ_NUM, 
					TERM_NAME 
		  FROM SR0TDT 
		  WHERE TERM_NAME NOT IN ('NONE', 'DEFAULT RECORDS', 'ADMINISTRATIVE')
		 ) AS T
	WHERE CONVERT(INT, SUBSTRING(T.TERM_CD, 1,2)) <= (DATEPART(YY, GETDATE()))%100 
	ORDER BY T.TERM_SEQ_NUM DESC
